import React from "react";
import data from "./buttonData";
 
function Button() {
  return (
    <>
      {Object.keys(data).map((elem, idx) => (
        <script
          dangerouslySetInnerHTML={{ __html: elem.code }}
          key={`button${idx}`}
        />
      ))}
    </>
  );
}
 
export default Button;
